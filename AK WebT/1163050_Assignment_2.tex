  $$
  x_1 = [0,1,1],x_2 = [1,1,0],x_3 = [1,0,1], bias_{hidden} = [10, 20], bias_{out} = 20, 
  g(z) =
    \begin{cases}
      0   & \quad z \leq -10\\
      1   & \quad z \geq 10 \\
      0.5 & \quad \text{sonst}
  \end{cases}
  $$\par
  $$
  Input =
  \begin{pmatrix}
    0 & 1 & 1 \\
    1 & 1 & 0 \\
    1 & 0 & 1
  \end{pmatrix},
  Hidden = 
  \begin{pmatrix}
    -20 & 20 & -40 \\
    -40 & 0 & 0 
  \end{pmatrix},
  Hidden^T = 
  \begin{pmatrix}
    -20 & -40 \\
     20 & 0 \\
    -40 & 0
  \end{pmatrix}
  $$\par
  $$
  Input \times Hidden^T +  =
  \begin{pmatrix}
    0 & 1 & 1 \\
    1 & 1 & 0 \\
    1 & 0 & 1
  \end{pmatrix}
  \times
  \begin{pmatrix}
    -20 & -40 \\
     20 & 0 \\
    -40 & 0
  \end{pmatrix}
  =
  \begin{pmatrix}
    -20 & 0 \\
     0 & -40 \\
    -60 & -40
  \end{pmatrix}
  $$\par
  $$
  \begin{pmatrix}
    -20 & 0 \\
     0 & -40 \\
    -60 & -40
  \end{pmatrix}
  + bias_{hidden} =
  \begin{pmatrix}
    -20 + 10 & 0 + 20 \\
    0 + 10 & -40 + 20 \\
    -60 + 10 & -40 + 20
  \end{pmatrix}
  =
  \begin{pmatrix}
    -10 & 20 \\
     0 & -20 \\
    -60 & -20
  \end{pmatrix}
  $$\par
  $$
  g(
    \begin{pmatrix}
      -10 & 20 \\
       0 & -20 \\
      -60 & -20
    \end{pmatrix}
  ) \times bias_{out} =
  \begin{pmatrix}
    0 & 1 \\
    1 & 0 \\
    0 & 0
  \end{pmatrix}
  \times
  \begin{pmatrix}
    40 \\
    -40
  \end{pmatrix}
  =
  \begin{pmatrix}
    -40 \\
    40 \\
    0
  \end{pmatrix}
  $$\par
  $$
  g(
    \begin{pmatrix}
      -40 \\
      40 \\
      0
    \end{pmatrix}
    + 20
  ) =
  g(
    \begin{pmatrix}
      -20 \\
      60 \\
      20
    \end{pmatrix}
  ) =
